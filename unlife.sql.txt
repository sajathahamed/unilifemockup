-- announcements
CREATE TABLE announcements (
    id uuid PRIMARY KEY,
    created_by uuid,
    target_audience text,
    body text,
    title text,
    created_at timestamp,
    updated_at timestamp
);

-- app_pages
CREATE TABLE app_pages (
    id uuid PRIMARY KEY,
    path text,
    sort_order integer,
    icon text,
    role text,
    label text
);

-- assignment_submissions
CREATE TABLE assignment_submissions (
    id uuid PRIMARY KEY,
    assignment_id uuid,
    student_id uuid,
    submitted_at timestamp,
    grade numeric,
    content text,
    feedback text,
    status text
);

-- assignments
CREATE TABLE assignments (
    id uuid PRIMARY KEY,
    created_at timestamp,
    updated_at timestamp,
    created_by uuid,
    course_id uuid,
    title text,
    description text,
    due_date date
);

-- courses
CREATE TABLE courses (
    id uuid PRIMARY KEY,
    lecturer text,
    colour text,
    course_name text,
    course_code text
);

-- deliveries
CREATE TABLE deliveries (
    id uuid PRIMARY KEY,
    order_id uuid,

);

);
